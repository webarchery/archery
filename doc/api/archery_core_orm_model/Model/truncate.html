static Future<bool> truncate<T extends Model>({Disk disk = Model.defaultDisk}) async {
  switch (disk) {
    case Disk.file:
      try {
        return await JsonFileModel.truncate<T>();
      } catch (e) {
        return false;
      }
    case Disk.sqlite:
      try {
        return await SQLiteModel.truncate<T>();
      } catch (e) {
        return false;
      }
    case Disk.s3:
      return false;
  }
}
