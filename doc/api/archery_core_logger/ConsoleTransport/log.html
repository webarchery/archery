@override
Future<void> log(LogEntry entry) async {
  final config = App().container.make<AppConfig>();
  final debugMode = config.get('app.debug', false);

  // Suppress all logs unless debug mode is enabled
  if (!debugMode) return;

  if (formatJson) {
    _logJson(entry);
  } else {
    _logFormatted(entry);
  }
}
