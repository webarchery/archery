static dynamic sortRecursive(dynamic item) {
  if (item is Map) {
    final sortedKeys = item.keys.toList()..sort();
    return {
      for (var key in sortedKeys) key: sortRecursive(item[key]),
    };
  } else if (item is List) {
    return item.map(sortRecursive).toList();
  }
  return item;
}
